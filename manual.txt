# ─────────────────────────────────────────
# DHCP — Probar que el servidor reparte IPs
# ─────────────────────────────────────────
docker compose up -d dhcp dhcp_client_demo   # levanta solo el DHCP y un cliente de prueba

docker compose logs -f dhcp_client_demo
# Deberías ver algo así:
#   --- IP asignada por Docker al iniciar ---     -> ej. 172.30.0.3 (la dio el IPAM de Docker)
#   --- Solicitando IP real por DHCP ---           -> udhcpc manda DISCOVER/REQUEST
#   --- IP obtenida del servidor DHCP ---          -> ej. 172.30.0.1xx (la dio dhcpd, NO Docker)

docker compose logs dhcp
# Acá se ven los DHCPDISCOVER / DHCPOFFER / DHCPREQUEST / DHCPACK que procesó el server

# Para ver el handshake completo a nivel de paquetes (opcional, para la defensa):
docker compose exec dhcp tcpdump -i eth0 -n port 67 or port 68

# Para ver las concesiones (leases) otorgadas:
docker compose exec dhcp cat /var/lib/dhcp/dhcpd.leases
